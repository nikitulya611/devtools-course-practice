#############################################
##### Testing
#############################################

set(prefix "${MODULE}")

add_test(
    NAME ${prefix}_can_Run
    COMMAND ${APPLICATION}
)
set_tests_properties (${prefix}_can_Run PROPERTIES
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_小an_Print_Help_Without_Arguments
    COMMAND ${APPLICATION}
)

set_tests_properties (${prefix}_小an_Print_Help_Without_Arguments PROPERTIES
    PASS_REGULAR_EXPRESSION "This is an application to demonstrate the operation of the binary search algorithm"
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_小an_Check_2_Arguments
    COMMAND ${APPLICATION} 1 1
)

set_tests_properties (${prefix}_小an_Check_2_Arguments PROPERTIES
    PASS_REGULAR_EXPRESSION "Error occured: "
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Check_Wrong_Argument_Type
    COMMAND ${APPLICATION} as
)

set_tests_properties (${prefix}_Check_Wrong_Argument_Type PROPERTIES
    PASS_REGULAR_EXPRESSION "Error occured: Wrong argument type."
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Returns_Correct_Index
    COMMAND ${APPLICATION} 3
)

set_tests_properties (${prefix}_Returns_Correct_Index PROPERTIES
    PASS_REGULAR_EXPRESSION "1"
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Returns_Minus_One
    COMMAND ${APPLICATION} 2
)

set_tests_properties (${prefix}_Returns_Minus_One PROPERTIES
    PASS_REGULAR_EXPRESSION "-1"
    LABELS "${MODULE}")
